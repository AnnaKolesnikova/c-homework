#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>
#include<conio.h>
#include<stdlib.h>
#include<ctype.h>
#include<String.h>

int main()
{
	char x[40] = "hghghg 6454645 ggjhj56 5656 jhjhj";
	char * p;
	int k = 0, o = 0;

	//printf("\nVveditr stroku x:  ");
	//fflush(stdin);
	//gets_s(x);

	p = strtok(x, " ");

	while(p != NULL)
	{
		printf("Leksema p : %s\n", p);
		printf("Kolichestvo simvolov v lekseme: %s ravno %d\n", p, strlen(p));
		printf("Soderjimoe stroki x: %s\n", x);
		

		//работаем с выделенной лексемой(подстрокой)
		for(int j = 0; j < strlen(p); j++)
		{
			if (p[j] >= 48 && p[j] <= 57)
			{
				k++;
				if(k == strlen(p))
				{
					o++;
					printf("\no = %d", o);
				}
			}
		}

		printf("\n");
		//printf("\no = %d", o);
		printf("\n---------------------------------------\n\n");
		p = strtok(NULL, " ");
	}

	for(int i = 0; i < 40; i++)
	{
		printf("%-4c", x[i]);
	}

	

	printf("\n\n\nThe end!!!\n");
	_getch();
	return 0;
}
