int main {
	
int x, y;
	
	
x = 21;
	
	
y = 8;
	
	
printf("%d+%d=%d\n",x,y,x+y);
	// 21 + 8 = 29
	
printf("%d-%d=%d\n",x,y,x-y);
	// 21 - 8 = 13
	
printf("%d*%d=%d\n",x,y,x*y);
	// 21 * 8 = 168
	
printf("%d/%d=%d\n",x,y,x/y);	// 21 / 8 = 2
	
return 0;

}
