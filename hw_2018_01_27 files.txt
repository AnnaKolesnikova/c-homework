#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h> //директива препроцессора
#include<conio.h>

void main()
{
	int account;
	char name[30];
	double balance;

	FILE *cfPtr; //указатель на тип FILE
	if ((cfPtr = fopen("D:\\clents.dat", "r")) == NULL)
	{
		printf("\nFile could not be opened\n");
		return;
	}
	printf("%-10s%-13s%-17s\n", "Account", "Name", "Balance"); //заголовок
	fscanf(cfPtr, "%d%s%lf", &account, name, &balance);

	while (!feof(cfPtr))
	{
		printf("%-10d%-13s%7.2lf\n", account, name, balance);
		fscanf(cfPtr, "%d%s%lf", &account, name, &balance);
	}
	_getch();
	fclose(cfPtr);
}
